一、web网卡列表
1.发送：
消息ID：1                                
消息内容：
{
“netlist”:{                                     // 网卡列表
"net": [{                      // 网卡条件（空是所有网卡，可以指定网卡条件搜索未实现）
}]
}
}
2.接收：
消息ID：1
消息内容：
{
"netlist": {                                                       // 网卡列表
                   "code": 1,                                              // 返回结果0失败，1成功
“msg”:””,                                                        //成功为空，失败返回具体错误信息
                   "dns": [{
                            "name": "8.8.7.8"                      // 首选DNS
                   }, {
                            "name": "8.8.8.6"                      // 备用DNS
                   }],
                   "net": [{
                            "gateway": "193.168.14.1",   // 网关
                            "ip": "193.168.14.101",           // IP地址
                            "mask": "255.255.255.0",       // 子网掩码
                            "name": "M0",                           // 网卡名称，不可修改
"status":"1",                               //网卡状态（1:UP，0:DOWN）
                            "type": "static"                           // 网络类型（static）
                   }, {
                            "gateway": "",
                            "ip": "10.0.0.1",
                            "mask": "255.255.255.0",

"status":"0",

         }
}
